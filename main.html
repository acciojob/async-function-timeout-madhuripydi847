document.getElementById("btn").addEventListener("click", async function () { const textInput = document.getElementById("text").value; const delayInput = parseInt(document.getElementById("delay").value, 10); const outputDiv = document.getElementById("output"); // Clear previous output outputDiv.textContent = ""; if (!textInput || isNaN(delayInput) || delayInput < 0) { outputDiv.textContent = "Please enter valid text and delay time."; return; } // Function to introduce delay using Promise function delay(ms) { return new Promise(resolve => setTimeout(resolve, ms)); } // Await the delay and then display the message await delay(delayInput); outputDiv.textContent = textInput; });
