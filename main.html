document.getElementById("btn").addEventListener("click", async function () { const text = document.getElementById("text").value; const delay = parseInt(document.getElementById("delay").value); if (!text || isNaN(delay) || delay < 0) { alert("Please enter a valid message and a non-negative delay."); return; } await delayMessage(delay); document.getElementById("output").innerText = text; }); function delayMessage(ms) { return new Promise(resolve => setTimeout(resolve, ms)); }
